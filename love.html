给特别的人

有人说，最浪漫的事就是把爱写进代码里。

因为代码不会说谎，永远忠实执行写下的每一行逻辑。

就像我的心情，清晰、确定、不可动摇。

const heart = {
    owner: "你",
    beats: "forever",
    rhythm: ["ba-dum", "ba-dum"],
    reason: "because of you",
    
    beatFor(person) {
        if (person === "你") {
            return "💓ba-dum ba-dum💓";
        }
        return "still...";
    },
    
    getPromise() {
        return new Promise((resolve) => {
            resolve("❤️ 我会一直在这里 ❤️");
        });
    }
};

// 我的心跳节奏
for (let moment of heart.beats) {
    console.log(heart.beatFor("你"));
}

// 这份承诺
heart.getPromise().then(message => {
    console.log(message);
});
